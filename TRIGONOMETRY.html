(TRIGONOMETRY)

AC=h (hypotenuse)

AB=p(perpendicular)

BC=b(base)
Sinθ=AB/AC=p/h;

Cosθ=BC/AC=b/h;

tanθ=AB/BC=p/b;

Cotθ=BC/AB=b/p;

secθ=AC/BC=h/b;

cosecθ=AC/AB=h/p;

sinθ×cosecθ=1

secθ×cosθ=1

tanθ×cotθ=1

(sinθ)2+(cosθ)2=1

sin2θ=1-cos2θ

cos2θ=1-sin2θ

sec2θ-tan2θ=1

sec2θ=1+tan2θ

tan2θ=sec2θ-1

cosec2θ-cot2θ=1

cosec2θ=1+cot2θ

cot2θ=cosec2θ-1

sin(-θ)=-sinθ

cos(-θ)=cosθ

tan(-θ)=-tanθ

cot(-θ)=-cotθ

sec(-θ)=secθ

cosec(-θ)=-cosecθ
